### 5.3 JavaScript配置：
```javascript
// 淘宝商品链接常量
const TAOBAO_URL = 'https://item.taobao.com/item.htm?ft=t&id=1034939670570';

// 页面元素
const taobaoBtn = document.getElementById('taobao-btn');
const copyLinkBtn = document.getElementById('copy-link-btn');

// 淘宝下单按钮 - 直接跳转
taobaoBtn.addEventListener('click', function() {
    // 直接跳转到淘宝商品页
    window.open(TAOBAO_URL, '_blank');
});

// 复制商品链接按钮
copyLinkBtn.addEventListener('click', async function() {
    try {
        await navigator.clipboard.writeText(TAOBAO_URL);
        // 显示复制成功反馈
        showNotification('淘宝商品链接已复制到剪贴板', 'success');
    } catch (err) {
        // 备用复制方法
        showNotification('复制失败，请手动复制链接：' + TAOBAO_URL, 'error');
    }
});
```
